import type { Request, Response } from "express";
import express from "express";
import { eq } from "drizzle-orm";
import { db } from "../db";
import { claims, claimPii, supplements } from "../db/schema";
import { applyPiiMasking, canViewFullPii, Role } from "../utils/piiMasking";
import { toCsv } from "../utils/csv";
import { logExport } from "../utils/audit";

// PDF generator placeholder
// Use pdfkit or @react-pdf/renderer later. This keeps the route stable now.
async function renderClaimPdf(args: {
  exportType: "intelligence_summary" | "claim_packet_masked" | "claim_packet_unmasked";
  claim: any;
  supplements: any[];
  generatedBy: { userId: string; role: Role; tenantId: string };
}) {
  const lines: string[] = [];

  lines.push("ClaimSignal Export");
  lines.push(`Export Type: ${args.exportType.toUpperCase()}`);
  lines.push(`Generated By User: ${args.generatedBy.userId}`);
  lines.push(`Generated At: ${new Date().toISOString()}`);
  lines.push("");

  lines.push(`Claim ID: ${args.claim.id}`);
  lines.push(`Claim Status: ${args.claim.claimStatus ?? ""}`);
  lines.push(`Friction Score: ${args.claim.frictionScore ?? ""}`);
  lines.push(`Escalation Category: ${args.claim.escalationCategory ?? ""}`);
  lines.push(`Approval Probability: ${args.claim.approvalProbability ?? ""}`);
  lines.push("");

  if (args.exportType !== "intelligence_summary") {
    lines.push(`Homeowner Name: ${args.claim.homeownerName ?? ""}`);
    lines.push(`Property Address: ${args.claim.propertyAddress ?? ""}`);
    lines.push(`Claim Number: ${args.claim.claimNumber ?? ""}`);
    lines.push(`Policy Number: ${args.claim.policyNumber ?? ""}`);
    lines.push("");
  }

  lines.push("Supplements");
  for (const s of args.supplements) {
    lines.push(
      `Submitted: ${s.submittedDate ?? ""} Amount: ${s.supplementAmount ?? ""} Approved: ${s.approvedAmount ?? ""} Status: ${s.status ?? ""}`
    );
  }

  // Return a simple text based PDF substitute as bytes for now.
  // Replace with real PDF rendering next. Route stays identical.
  return Buffer.from(lines.join("\n"), "utf8");
}

function requireAuth(req: any) {
  if (!req.user) throw new Error("UNAUTHORIZED");
  return req.user as { id: string; role: Role; tenantId: string };
}

const router = express.Router();

router.get("/exports/claims/:claimId", async (req: Request, res: Response) => {
  try {
    const user = requireAuth(req);
    const claimId = req.params.claimId;

    const format = (req.query.format as string) === "csv" ? "csv" : "pdf";
    const exportTypeRaw = (req.query.type as string) || "claim_packet_masked";

    const exportType =
      exportTypeRaw === "intelligence_summary"
        ? "intelligence_summary"
        : exportTypeRaw === "claim_packet_unmasked"
          ? "claim_packet_unmasked"
          : "claim_packet_masked";

    const unmaskedRequested = exportType === "claim_packet_unmasked";
    const privileged = canViewFullPii(user.role);

    // Force masked if not privileged
    const effectiveExportType =
      unmaskedRequested && privileged ? "claim_packet_unmasked" : exportType === "intelligence_summary" ? "intelligence_summary" : "claim_packet_masked";

    // Tenant scope
    const claimRow = await db
      .select()
      .from(claims)
      .where(eq(claims.id, claimId))
      .limit(1);

    if (claimRow.length === 0) return res.status(404).send("Not found");

    const claim = claimRow[0];

    if (user.role !== "super_admin" && claim.tenantId !== user.tenantId) {
      return res.status(403).send("Forbidden");
    }

    // Join PII only when needed
    let merged: any = { ...claim };

    if (effectiveExportType !== "intelligence_summary") {
      const piiRow = await db.select().from(claimPii).where(eq(claimPii.claimId, claimId)).limit(1);
      if (piiRow.length > 0) merged = { ...merged, ...piiRow[0] };
    }

    // Supplements
    const supps = await db.select().from(supplements).where(eq(supplements.claimId, claimId));

    // Apply masking unless unmasked and privileged
    if (effectiveExportType === "claim_packet_masked") {
      merged = applyPiiMasking(merged, user.role);
    }

    // Audit log
    await logExport({
      actorUserId: user.id,
      tenantId: user.tenantId,
      claimId,
      format,
      exportType: effectiveExportType,
    });

    if (format === "csv") {
      // CSV columns
      const columns =
        effectiveExportType === "intelligence_summary"
          ? [
              { key: "id", label: "Claim ID" },
              { key: "claimStatus", label: "Status" },
              { key: "frictionScore", label: "Friction Score" },
              { key: "escalationCategory", label: "Escalation Category" },
              { key: "approvalProbability", label: "Approval Probability" },
            ]
          : [
              { key: "id", label: "Claim ID" },
              { key: "homeownerName", label: "Homeowner" },
              { key: "propertyAddress", label: "Address" },
              { key: "claimNumber", label: "Claim Number" },
              { key: "policyNumber", label: "Policy Number" },
              { key: "claimStatus", label: "Status" },
              { key: "frictionScore", label: "Friction Score" },
              { key: "escalationCategory", label: "Escalation Category" },
              { key: "approvalProbability", label: "Approval Probability" },
            ];

      const csv = toCsv([merged], columns);
      res.setHeader("Content-Type", "text/csv; charset=utf-8");
      res.setHeader("Content-Disposition", `attachment; filename="claim_${claimId}_${effectiveExportType}.csv"`);
      return res.status(200).send(csv);
    }

    // PDF
    const pdfBytes = await renderClaimPdf({
      exportType: effectiveExportType,
      claim: merged,
      supplements: supps,
      generatedBy: { userId: user.id, role: user.role, tenantId: user.tenantId },
    });

    res.setHeader("Content-Type", "application/pdf");
    res.setHeader("Content-Disposition", `attachment; filename="claim_${claimId}_${effectiveExportType}.pdf"`);

    return res.status(200).send(pdfBytes);
  } catch (e: any) {
    if (String(e?.message) === "UNAUTHORIZED") return res.status(401).send("Unauthorized");
    return res.status(500).send("Export error");
  }
});

export default router;
